    3. Modify `publishContent(contentId)` — AFTER the existing SQL updates succeed (after `return true` is about to fire), add the hook/webhook/cache pipeline. Fetch content row first for payload data:
       ```
       // Fetch content details for hook payload
       const content = await this.db.prepare(
         'SELECT collection_id, title, slug FROM content WHERE id = ?'
       ).bind(contentId).first() as any

       // Fire AFTER_CONTENT_PUBLISH hook
       try {
         const hookSystem = getHookSystem()
         const hookPromise = hookSystem.execute(HOOKS.AFTER_CONTENT_PUBLISH, {
           contentId,
           collectionId: content?.collection_id,
           title: content?.title,
           slug: content?.slug,
         })
         if (this.ctx?.waitUntil) {
           this.ctx.waitUntil(hookPromise)
         } else {
           hookPromise.catch(err => console.error('[scheduler] publish hook error:', err))
         }

         // Outgoing webhook
         const webhookUrls = this.env?.WEBHOOK_URLS as string | undefined
         const webhookSecret = this.env?.WEBHOOK_SECRET as string | undefined
         if (webhookUrls) {
           const payload = {
             event: HOOKS.AFTER_CONTENT_PUBLISH,
             contentId,
             contentType: content?.collection_id,
             timestamp: Date.now(),
           }
           if (this.ctx?.waitUntil) {
             this.ctx.waitUntil(deliverWebhooks(webhookUrls, webhookSecret, payload))
           } else {
             deliverWebhooks(webhookUrls, webhookSecret, payload).catch(err =>
               console.error('[scheduler] publish webhook error:', err)
             )
           }
         }

         // Cache invalidation
         if (content?.collection_id) {
           const cache = getCacheService(CACHE_CONFIGS.api!)
           await cache.invalidate(`content:list:${content.collection_id}:*`)
           await cache.invalidate('content-filtered:*')
         }
       } catch (hookErr) {
         console.error('[scheduler] publish hooks/cache failed:', hookErr)
       }
       ```
       Place this block AFTER the workflow_state_id update and BEFORE `return true`.
